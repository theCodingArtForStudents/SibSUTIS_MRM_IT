300 {
301  if( message == WM_DESTROY ){
302  PostQuitMessage( EXIT_SUCCESS );
303  return true;
304  }
305  if(WM_SIZING == message || WM_SIZE == message) {
307  RECT rect;
308  if( ! GetWindowRect(hWnd, &rect) ){
310  return false;
311  }
312 
313  WSize oldSize(_width, _height);
314 
316  _x = rect.left;
317  _y = rect.top;
318  _width = rect.right-rect.left;
319  _height = rect.bottom - rect.top;
320 
321  WSize newSize(rect.right-rect.left, rect.bottom - rect.top);
322 
323  return this->event( new WResizeEvent(newSize, oldSize) );
324  }
325  if(WM_MOVE == message || WM_MOVING == message){
326  WPoint oldPos( _x, _y);
327 
328  RECT rect;
329  if( ! GetWindowRect(hWnd, &rect) ){
331  return false;
332  }
333 
334  WPoint newPos(rect.left, rect.top);
335 
336  return this->event( new WMoveEvent(newPos, oldPos) );
